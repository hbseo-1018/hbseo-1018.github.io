async function searchStats(button) {
    const section = $(button).closest('section');
    const sectionId = section.attr('id');
    const startDate = section.find('.start-date').val();
    const endDate = section.find('.end-date').val();

    const handlers = {
        'summary-stats': renderSummaryStats,
        'channel-stats': renderChannelStats,
        'member-stats': renderMemberStats,
    };

    if (!handlers[sectionId]) {
        return;
    }

    blockView(true);
    await handlers[sectionId](startDate, endDate);
    blockView(false);
}

module.exports.getStatistics = async (req, res) => {
    const { startDate, endDate } = req.body;

    try {
        const users = await statsDao.findActiveUsers();
        const rows = await statsDao.findStatsByPeriod(startDate, endDate);

        const result = {
            total: {
                data: {
                    groupA: {},
                    groupB: {}
                }
            }
        };

        users.forEach(user => {
            result[user.id] = {
                name: user.name,
                data: {
                    userNo: user.userNo,
                    total: 0
                }
            };
        });

        rows.forEach(row => {
            if (!result[row.userId]) {
                return;
            }

            const group = resolveGroup(result[row.userId], row.typeCode);
            result.total.data[group].total = (result.total.data[group].total || 0) + 1;
            result[row.userId].data.total += 1;
        });

        send(res, 200, result);
    } catch (error) {
        send(res, 500, {
            message: '통계 조회 중 오류가 발생했습니다.'
        });
    }
};

function validateDocumentForm(data) {
    if (!data.date) return { valid: false, message: '날짜를 입력하세요.' };
    if (!data.receiver) return { valid: false, message: '수신처를 입력하세요.' };
    if (!data.address) return { valid: false, message: '주소를 입력하세요.' };
    if (!data.title) return { valid: false, message: '제목을 입력하세요.' };
    if ((data.items || []).length > 20) return { valid: false, message: '상세내역은 최대 20개까지 입력 가능합니다.' };

    return { valid: true };
}

module.exports.downloadDocumentPdf = async function(req, res) {
    try {
        const documentId = req.params.documentId;
        const payload = JSON.parse(req.query.data || '{}');
        const pdfPath = await reportService.createDocumentPdf(payload);

        res.setHeader('Content-Disposition', `attachment; filename=document_${documentId}.pdf`);
        res.setHeader('Content-Type', 'application/octet-stream');

        fs.createReadStream(pdfPath)
            .on('close', () => fs.unlink(pdfPath, () => {}))
            .pipe(res);

        logService.addLog(req.session.userId, 'download_document_pdf', {
            documentId,
            reason: req.query.reason
        });
    } catch (error) {
        send(res, 400, { message: 'PDF 생성 중 오류가 발생했습니다.' });
    }
};

module.exports.addTargetsByDomain = async function(documentId, groupId, filters) {
    const db = await getDb({ transaction: true });

    try {
        const domains = await targetDao.findDomains(db, groupId);
        if (!domains.length) {
            await db.commit();
            return true;
        }

        const targets = await targetDao.findOpenTargets(db, {
            domains,
            ageType: filters.ageType,
            numberType: filters.numberType,
            statusCode: filters.statusCode
        });

        const rows = targets
            .filter(target => matchesNumberFilter(target.historyList, filters.numberType))
            .map(target => [
                documentId,
                target.caseId,
                target.targetId,
                filters.numberType
            ]);

        if (rows.length) {
            await documentDao.addTargetRows(db, rows);
        }

        await db.commit();
        return true;
    } catch (error) {
        await db.rollback();
        throw error;
    }
};

function matchesNumberFilter(historyJson, numberType) {
    if (!numberType || numberType === 'all') {
        return true;
    }

    const historyList = JSON.parse(historyJson || '[]');
    const hasCaseNumber = historyList.some(item => !StringUtils.isBlank(item.caseNo));

    return numberType === 'exists' ? hasCaseNumber : !hasCaseNumber;
}

-- 문서에 연결된 URL 목록을 조회하면서 사건번호, 담당자, 유형, JSON 필드, URL을 통합 검색합니다.
SELECT t.target_id,
       t.document_id,
       t.case_id,
       c.case_no,
       c.support_type,
       c.age_type,
       c.status_code,
       c.history_json,
       COALESCE(s.url, m.url) AS url,
       managers.owner_names,
       managers.worker_names
FROM DocumentTarget t
LEFT JOIN CaseInfo c
       ON t.case_id = c.case_id
LEFT JOIN SiteTarget s
       ON t.site_target_id = s.site_target_id
LEFT JOIN ManualTarget m
       ON t.manual_target_id = m.manual_target_id
LEFT JOIN ManagerSummary managers
       ON managers.case_id = c.case_id
WHERE t.document_id = :documentId
  AND (
       c.case_no LIKE :keyword
    OR managers.owner_names LIKE :keyword
    OR managers.worker_names LIKE :keyword
    OR c.support_type LIKE :keyword
    OR c.age_type LIKE :keyword
    OR JSON_EXTRACT(c.history_json, '$[*].dept') LIKE :keyword
    OR JSON_EXTRACT(c.history_json, '$[*].caseNo') LIKE :keyword
    OR c.status_code LIKE :keyword
    OR COALESCE(s.url, m.url) LIKE :keyword
  )
ORDER BY c.case_no
LIMIT :offset, :pageSize;

// 엑셀을 임시 저장한 뒤 Sheet1을 읽고, 검증 실패 데이터는 DB 저장 전에 화면으로 돌려줍니다.
module.exports.uploadCompletionExcel = async function(req, res) {
    const documentId = req.params.documentId;
    const tempPath = fileService.createTempPath('xlsx');

    try {
        await uploadService.saveStream(req, tempPath, {
            maxSize: 10 * 1024 * 1024
        });

        const workbook = XLSX.read(tempPath, { type: 'file' });
        const sheet = workbook.Sheets['Sheet1'];

        if (!sheet) {
            return send(res, 400, { message: 'Sheet1을 찾을 수 없습니다.' });
        }

        const rows = XLSX.utils.sheet_to_json(sheet, { raw: false });
        const errors = validateRows(rows);

        if (errors.length) {
            return send(res, 200, {
                success: false,
                failData: errors
            });
        }

        for (const row of rows) {
            const caseInfo = await caseDao.findByCaseNo(row['사건번호']);
            if (!caseInfo) {
                row.error = '입력한 사건번호에 해당하는 이력이 없습니다.';
                errors.push(row);
                continue;
            }

            await documentDao.addCompletion(documentId, {
                caseId: caseInfo.caseId,
                url: row['URL'].trim(),
                numberType: row['사건번호 유무']
            });
        }

        send(res, 200, {
            success: errors.length === 0,
            count: rows.length,
            failData: errors
        });
    } finally {
        fs.unlink(tempPath, () => {});
    }
};

function validateRows(rows) {
    return rows.reduce((errors, row) => {
        if (!row['사건번호'] || !isValidCaseNo(row['사건번호'])) {
            row.error = '사건번호 형식이 잘못되었습니다.';
            errors.push(row);
        } else if (!row['URL'] || !row['URL'].trim()) {
            row.error = 'URL 정보가 잘못되었습니다.';
            errors.push(row);
        } else if (row['사건번호 유무'] && !['1', '2'].includes(String(row['사건번호 유무']))) {
            row.error = '사건번호 유무 정보가 잘못되었습니다.';
            errors.push(row);
        }

        return errors;
    }, []);
}

클라이언트가 Form 제출 전 서버에서 발급받은 일회성 토큰과 AES Key를 이용해 비밀번호를 암호화하고, 서버는 전달받은 토큰으로 Redis에 저장된 Key를 조회해 복호화한 뒤 기존 로그인 검증 로직을 수행하도록 개선했습니다.

토큰 생성, AES Key 생성, Redis 저장, 암복호화 처리 

/*
1. crypto.randomBytes 기반 64자리 토큰 및 256bit AES Key 생성
2. Redis를 활용해 토큰-Key 매핑 정보를 5분 TTL로 저장
3. 클라이언트에서 CryptoJS AES-CBC + PKCS7 방식으로 비밀번호 암호화
4. 서버에서 토큰 유효성, 형식, 만료 여부를 검증한 뒤 AES 복호화 수행
5. 로그인 성공 후 사용된 토큰 삭제로 재사용 위험 감소
6. 토큰 누락, 형식 오류, 만료, 복호화 실패 등 예외 상황별 에러 코드 분리
*/

generateToken() {
    return crypto.randomBytes(32).toString('hex');
}

generateAesKey() {
    return crypto.randomBytes(32).toString('hex');
}

createTokenKeyPair() {
    return new Promise((resolve, reject) => {
        const token = this.generateToken();
        const aesKey = this.generateAesKey();

        const tokenData = JSON.stringify({
            key: aesKey,
            createdAt: Date.now()
        });

        this.redisClient.setex(`token:${token}`, this.tokenTTL, tokenData, (err) => {
            if (err) reject(err);
            else resolve(token);
        });
    });
}

        const password = await tokenService.decrypt(req.body.password, req.body.token);

            errorCode: error.errorCode || 'DECRYPT_FAIL'
        });
    }
};

function encryptPassword(password, aesKey) {
    const keyBytes = CryptoJS.enc.Base64.parse(aesKey);
    const iv = CryptoJS.lib.WordArray.random(16);

    const encrypted = CryptoJS.AES.encrypt(password, keyBytes, {
        iv: iv,
        mode: CryptoJS.mode.CBC,
        padding: CryptoJS.pad.Pkcs7
    });

    const ivHex = iv.toString(CryptoJS.enc.Hex);
    const encryptedBase64 = encrypted.toString();

    return ivHex + ':' + encryptedBase64;
}
