/*
1. 외부 시스템이 accessKey를 포함해 /api/sample/v1/items 호출
2. 인증 테이블에서 Key 유효성 검증
3. 인증 실패 시 401 응답과 실패 로그 저장
4. 인증 성공 시 최근 데이터 조회
5. 결과를 JSON 배열로 변환해 응답
6. 처리 결과를 별도 로그 테이블에 저장
*/

@ResponseBody
@PostMapping("/api/sample/v1/items")
public String getItems(@RequestBody Map<String, String> requestBody,
                       HttpServletRequest request,
                       HttpServletResponse response) {

    JSONObject result = new JSONObject();
    AuditLog log = new AuditLog();
    log.setClientIp(ClientIpUtil.get(request));
    log.setRequestText(requestBody.toString());

    try {
        String accessKey = requestBody.get("accessKey");
        KeyInfo keyInfo = dataService.findAccessKey(accessKey);

        if (keyInfo == null) {
            response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
            return fail(result, log, "AUTH_FAIL", "허용되지 않은 API KEY입니다.", null);
        }

        log.setRequester(keyInfo.getRequesterType());

        List<DataItem> items = dataService.findRecentItems();
        if (items.isEmpty()) {
            response.setStatus(HttpServletResponse.SC_NO_CONTENT);
            return fail(result, log, "EMPTY_DATA", "데이터가 존재하지 않습니다.", null);
        }

        JSONArray data = new JSONArray();
        JSONArray logData = new JSONArray();
        DateTimeFormatter dateFormat = DateTimeFormatter.ofPattern("yyyy-MM-dd");

        for (DataItem item : items) {
            JSONObject row = new JSONObject();
            row.put("SEQ", item.getSeq());
            row.put("ITEM_ID", item.getItemId());
            row.put("ITEM_TYPE", item.getItemType());
            row.put("ITEM_VALUE", item.getItemValue());
            row.put("CREATED_AT", dateFormat.format(item.getCreatedAt()));
            data.add(row);

            JSONObject logRow = new JSONObject();
            logRow.put("ITEM_ID", item.getItemId());
            logRow.put("ITEM_VALUE", item.getItemValue());
            logData.add(logRow);
        }

        result.put("status", 1);
        result.put("message", "SUCCESS");
        result.put("data", data);

        response.setStatus(HttpServletResponse.SC_OK);
        log.setType("API_CALL_SUCCESS");
        log.setResponseText(logData.toString());
        auditService.addLog(log);
        return result.toString();

    } catch (Exception e) {
        response.setStatus(HttpServletResponse.SC_UNPROCESSABLE_ENTITY);
        return fail(result, log, "SYSTEM_ERROR", "알 수 없는 오류입니다. 시스템에 문의 바랍니다.", e);
    }
}

private String fail(JSONObject result, AuditLog log, String type, String message, Exception e) {
    result.put("status", -1);
    result.put("message", message);
    result.put("data", null);

    log.setType(type);
    if (e != null) {
        log.setErrorText(e.getMessage());
    }
    auditService.addLog(log);
    return result.toString();
}

@Transactional(isolation=Isolation.READ_COMMITTED, rollbackFor = {Exception.class})
@Override
public void start() {
    long startedAt = System.currentTimeMillis();
    LocalDateTime now = LocalDateTime.now();
    String endDate = now.format(DateTimeFormatter.ofPattern("yyyyMMddHHmmss"));
    String startDate = now.minusDays(1).format(DateTimeFormatter.ofPattern("yyyyMMdd000000"));

    List<Integer> targetIds = itemDao.findTargetIds(startDate, endDate);
    if (targetIds == null || targetIds.isEmpty()) {
        return;
    }

    Map<String, String> resultLog = new HashMap<String, String>();

    for (int targetId : targetIds) {
        try {
            int savedCount = itemDao.countSavedItems(targetId);

            if (savedCount <= 0) {
                itemService.addItem(targetId);
            } else {
                itemService.updateItem(targetId);
            }
        } catch (Exception e) {
            if (resultLog.size() < 10) {
                resultLog.put("error:" + targetId, e.getClass().getSimpleName());
            }
        }
    }

    resultLog.put("count", String.valueOf(targetIds.size()));
    resultLog.put("elapsedMs", String.valueOf(System.currentTimeMillis() - startedAt));
    logService.addLog(null, "batch_processed_result", "127.0.0.1", resultLog);
}

@Override
public void upsertItem(int targetId) throws Exception {
    if (!itemDao.existsTarget(targetId)) {
        return;
    }

    int savedCount = itemDao.countSavedItems(targetId);
    if (savedCount <= 0) {
        itemService.addItem(targetId);
    } else {
        itemService.updateItem(targetId);
    }
}

private void addExtractedItems(int targetId) throws Exception {
    if (targetId <= 0) {
        return;
    }

    SourceRecord source = sourceService.findSource(targetId);
    validateSource(source);

    OutputRecord baseItem = new OutputRecord();
    baseItem.setTargetId(targetId);
    baseItem.setUserName(source.getUserName());
    baseItem.setOwnerName(source.getOwnerName());
    baseItem.setRequestUserId(source.getRequestUserId());
    baseItem.setCreatedAt(source.getClosedAt());
    baseItem.setStatus(resolveStatus(source));

    if (StringUtils.isBlank(baseItem.getStatus())) {
        return;
    }

    PersonInfo mainPerson = personService.findMainPerson(targetId);
    if (mainPerson == null || StringUtils.isBlank(mainPerson.getMobileNo())) {
        throw new CommonException(20);
    }
    baseItem.setMainMobileNo(mainPerson.getMobileNo());

    JSONObject answerJson = source.getAnswerJson();
    addAnswerItems(baseItem, answerJson, "ACCOUNT_GROUP", "value", "ACCOUNT");
    addAnswerItems(baseItem, answerJson, "WALLET_GROUP", "value", "WALLET");

    List<PersonInfo> people = personService.findRelatedPeople(targetId);
    if (people == null) {
        return;
    }

    for (PersonInfo person : people) {
        if (person == null) {
            continue;
        }

        addItem(baseItem, "EMAIL", person.getEmail(), "-");
        addItem(baseItem, "PHONE", person.getPhoneNo(), regex);
        addItem(baseItem, "MOBILE", person.getMobileNo(), regex);
    }
}

private void addAnswerItems(OutputRecord baseItem, JSONObject answerJson,
                            String answerKey, String valueKey, String type) {
    if (answerJson == null) {
        return;
    }

    JSONArray rows = (JSONArray) answerJson.get(answerKey);
    if (rows == null || rows.isEmpty()) {
        return;
    }

    for (Object row : rows) {
        JSONObject object = (JSONObject) row;
        String value = (String) object.get(valueKey);
        addItem(baseItem, type, value, regex);
    }
}

private void addItem(OutputRecord baseItem, String type, String value, String removePattern) {
    if (StringUtils.isBlank(value)) {
        return;
    }

    OutputRecord item = baseItem.copy();
    item.setItemType(type);
    item.setItemValue(value);
    item.setSearchValue(value.replaceAll(removePattern, ""));
    itemDao.addItem(item);
}
대용량 암호화 컬럼 검색 성능 개선

계좌번호와 은행 정보가 암호화된 컬럼으로 저장되어 있었고, 해당 컬럼을 조건으로 조회하는 쿼리에서 인덱스가 제대로 활용되지 않아 성능 저하가 발생했습니다. 데이터가 100만 건 이상으로 증가하면서 특정 계좌 이력 조회 시 응답 시간이 약 30초 이상 소요되었습니다.

기존 방식 

SELECT MainRecord.RECORD_ID AS RECORD_ID,
       RecordHistory.HISTORY_ID AS HISTORY_ID
       ...
FROM RecordHistory
INNER JOIN MainRecord
   ON MainRecord.VALUE_TEXT = RecordHistory.VALUE_TEXT
  AND MainRecord.TYPE_CODE = RecordHistory.TYPE_CODE
WHERE MainRecord.VALUE_TEXT = HEX(AES_ENCRYPT(#{valueText}, ${encryptKey}))
  AND MainRecord.TYPE_CODE = HEX(AES_ENCRYPT(#{typeCode}, ${encryptKey}))

정규화한 계좌번호와 은행 정보를 하나의 문자열로 조합한 뒤, 이를 MD5 해시값으로 변환한 신규 컬럼 HASH_VALUE를 추가했습니다.
이후 메인/이력 테이블 양쪽에 동일한 해시 컬럼을 저장하고 인덱스를 생성하여 조회 및 JOIN 조건을 단순화했습니다.
적용 후, 조회 쿼리 응답 시간을 30초에서 5초 미만으로 약 80% 이상 단축했습니다.

개선 쿼리 

SELECT r.RECORD_ID AS RECORD_ID,
       h.HISTORY_ID AS HISTORY_ID
       ...
FROM MainRecord r
JOIN RecordHistory h
  ON r.HASH_VALUE = h.HASH_VALUE
WHERE r.HASH_VALUE = #{hashValue}

@Override
public void syncData() {
    try {
        String mongoUrl = String.format("mongodb://%s:%s@%s/%s",
                mongoUser, mongoPassword, mongoHost, mongoDbName);
        MongoClient mongoClient = new MongoClient(new MongoClientURI(mongoUrl));

        mongoTemplate = new MongoTemplate(mongoClient, mongoDbName);
        MappingMongoConverter converter = (MappingMongoConverter) mongoTemplate.getConverter();
        converter.setTypeMapper(new DefaultMongoTypeMapper(null));

        List<String> targetIds = findSyncTargets();

        for (String targetId : targetIds) {
            SyncRecord current = findSourceRecord(Integer.parseInt(targetId));
            SyncRecord saved = findSavedRecord(current.getRecordId());

            List<String> addressRows = dataSyncDao.findAddressRows(Integer.parseInt(targetId));
            List<String> itemRows = dataSyncDao.findItemRows(Integer.parseInt(targetId));

            current.setAddressText(parseJoinedValue(addressRows, "address"));
            current.setItemText(parseJoinedValue(itemRows, "item"));

            if (StringUtils.isBlank(current.getStepCode())) {
                continue;
            }

            current.setStepName(resolveStepName(current.getStepCode()));

            String now = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss"));

            if (saved == null) {
                current.setCreatedAt(now);
                current.setUpdatedAt("");
                mongoTemplate.insert(current, "TB_SAMPLE_DATA");
            } else {
                current.setUpdatedAt(now);
                updateChangedFields(current, saved);
            }
        }

        logService.addLog(null, "sync_ok", "127.0.0.1", new HashMap<String, String>() {{
            put("sourceCount", String.valueOf(targetIds.size()));
        }});

    } catch (Exception e) {
        logService.addLog(null, "sync_fail", "127.0.0.1", Collections.singletonMap("error", e.getMessage()));
    }
}

private String parseJoinedValue(List<String> rows, String key) throws ParseException {
    if (rows == null || rows.isEmpty()) {
        return "";
    }

    List<String> values = new ArrayList<String>();
    JSONParser parser = new JSONParser();

    for (String row : rows) {
        JSONObject object = (JSONObject) parser.parse(row);
        String value = (String) object.get(key);

        if (!StringUtils.isBlank(value)) {
            values.add(value);
        }
    }

    return String.join(",", values);
}
MongoDB 기존 문서 조회 및 변경분 업데이트 

public int updateChangedFields(SyncRecord current, SyncRecord saved) {

    Query query = new Query();
    query.addCriteria(Criteria.where("recordId").is(current.getRecordId()));

    Update update = new Update();
    int changedCount = 0;

    changedCount += setIfChanged(update, "ownerName", saved.getOwnerName(), current.getOwnerName());
    changedCount += setIfChanged(update, "managerName", saved.getManagerName(), current.getManagerName());
    changedCount += setIfChanged(update, "stepName", saved.getStepName(), current.getStepName());

    update.set("updatedAt", current.getUpdatedAt());

    if (changedCount > 0) {
        mongoTemplate.updateMulti(query, update, "TB_SAMPLE_DATA");
    }
    return changedCount;
}

private int setIfChanged(Update update, String fieldName, String before, String after) {
    String safeBefore = StringUtils.defaultString(before);
    String safeAfter = StringUtils.defaultString(after);

    if (safeBefore.equals(safeAfter)) {
        return 0;
    }

    update.set(fieldName, safeAfter);
    return 1;
}

public SyncRecord findSavedRecord(String recordId) {
    Query query = new Query();
    query.addCriteria(Criteria.where("recordId").is(recordId));

    return mongoTemplate.findOne(query, SyncRecord.class, "TB_SAMPLE_DATA");
}
파일 관리 시스템 성능 최적화

파일 업로드 및 다운로드 과정에서 생성되는 임시 파일이 서버에 누적되는 문제를 방지하기 위해 Spring Scheduler 기반의 자동 정리 작업을 구현했습니다.
임시 파일 저장 경로를 주기적으로 점검하고, 일정 시간이 지난 파일만 삭제하도록 처리하여 다운로드 중인 파일 삭제 위험을 줄였습니다.

스케줄러 코드 

/*
1. 서비스 내에서 파일 업로드, 다운로드, 엑셀 생성, 첨부파일 미리보기 등의 기능을 처리하는 과정에서 서버 로컬 디렉터리에 임시 파일이 생성
2. 초기에는 기능 동작 자체에 집중되어 있어 생성된 임시 파일을 별도로 정리하는 프로세스 미구현
3. 이로 인해 사용자가 기능을 반복적으로 사용할수록 서버에 tmp 파일이 지속적으로 누적되는 문제 발생
4. 개별 기능마다 삭제 로직을 넣기보다는, 서버 내 임시 저장소를 주기적으로 점검하고 만료된 파일을 일괄 정리하는 방식이 더 안정적이라고 판단
5. 임시 파일 정리 로직을 Spring Scheduler 기반의 주기 작업으로 분리
*/
@Component
@RequiredArgsConstructor
@Slf4j
public class TempFileScheduler {

    @Value("${file.temp-path}")
    private String tempPath;

    @Scheduled(cron = "0 0 3 * * *")
    public void cleanTempFiles() {
        File tempDir = new File(tempPath);

        if (!tempDir.exists() || !tempDir.isDirectory()) {
            log.warn("임시 파일 경로가 존재하지 않거나 디렉터리가 아닙니다. path={}", tempPath);
            return;
        }

        File[] files = tempDir.listFiles();

        if (files == null || files.length == 0) {
            return;
        }

        long expireTime = System.currentTimeMillis() - Duration.ofDays(10).toMillis();

        for (File file : files) {
            if (file.isFile() && file.lastModified() < expireTime) {
                if (file.delete()) {
                    log.info("임시 파일 삭제 완료: {}", file.getAbsolutePath());
                } else {
                    log.warn("임시 파일 삭제 실패: {}", file.getAbsolutePath());
                }
            }
        }
    }
}
